CREATE KEYSPACE ChicagoErp  WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 3 };
CREATE TABLE IF NOT EXISTS ChicagoErp.Users (
    email varchar PRIMARY KEY,
    password varchar,
    salt varchar,
    id timeuuid,
    first_name varchar,
    middle_name varchar,
    last_name varchar,
    phone varchar,
    roles set<varchar>,
    extra_permissions set<varchar>);
